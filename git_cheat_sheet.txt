# get the current status
git status

# stage all changes
git add .

# unstage everything
git restore --staged .

# commit (can end with short message by: -m "msg" )
git commit
git commit -m "msg"

# push to remote repo
git push origin master

# search for a specific file throughout the git project history
git log --follow --all -- "**/psk.json"

# recursively list all files in current commit, with full paths, file sizes (bytes), sorted by 4th field / size, numeric, descending and aligned into columns.
git ls-tree -r -t -l --full-name HEAD | sort -k4 -nr | column -t

# find clutter in history. run command then read analysis file.
git filter-repo --analyze

# remove from history
git filter-repo --path "Sim Model" --invert-paths
git filter-repo --path-regex "PicoW/.+\.csv" --invert-paths

# add the remote repo to existing local repo (creating new remote after history change)
git remote add origin https://github.com/your-username/my-awesome-project.git

# verify remote repo
git remote --verbose

# first push to a new repo (setting the upstream repo)
git push --set-upstream origin master
